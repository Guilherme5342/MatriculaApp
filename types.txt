
type CargaHorariaCurso = {
    totalMinima: Number;
    totalObrigatoria: Number;
    optativaMinima: Number;
    eletivaMaxima: Number;
    maximaPorPeriodo: Number;
};

type Unidade = {
    codigo: String;
    nome: String;
};

type Localidade = {
    endereco: String;
};

type HorarioSemanal = {
    dia: String;
    horaInicio: String;
    horaFim: String;
    local: Localidade;
};

type Professor = {
    nome: String;
};

type CargaHorariaDisciplina = {
    total: Number;
    teorica: Number;
    pratica: Number;
};

type PrazoConclusao = {
    minimo: Number;
    medio: Number;
    maximo: Number;
};

type PeriodoLetivo = {
    ano: Number;
    numero: Number;
};

type Coordenador = {
    matricula: String;
    nome: String;
    email: String;
};

type Turma = {
    codigo: String;
    vagasOfertadas: Number;
    vagasOcupadas: Number;
    sede: String;
    horarios: HorarioSemanal[];
    professores: Professor[];
    periodo: PeriodoLetivo;
    disciplina: Disciplina;
};

type Curso = {
    codigo: String;
    nome: String;
    grau: String;
    modalidade: String;
    turno: String;
    coordenador: Coordenador;
    unidades: Unidade[];
    curriculos: EstruturaCurricular[];
};

type Aluno = {
    matricula: String;
    nome: String;
    email: String;
    ira: Number;
    curriculo: String;
    status: Boolean;
    periodoIngresso: PeriodoLetivo;
    curso: Curso;
};

type Matricula = {
    status: String;
    prioridade: Number;
    motivoIndeferimento: String;
    aluno: Aluno;
    turma: Turma;
};

type HistoricoAcademico = {
    aluno: Aluno;
    disciplinas: Disciplina[];
};

type Disciplina = {
    codigo: String;
    nome: String;
    modalidade: String;
    cargaHoraria: CargaHorariaDisciplina;
    unidade: Unidade;
    preRequisitos: Disciplina[];
};

type DisciplinaHistoricoAcademico = {
    status: String;
    mencao: String;
    periodo: PeriodoLetivo;
    disciplina: Disciplina;
};

type DisciplinaEstruturaCurricular = {
    tipo: String;
    nivel: Number;
    disciplina: Disciplina;
};

type EstruturaCurricular = {
    codigo: String;
    status: String;
    periodoLetivoEntradaVigor: PeriodoLetivo;
    disciplinas: Disciplina[];
    prazoConclusao: PrazoConclusao;
    cargaHoraria: CargaHorariaCurso;
};

enum TipoUsuario {
    ALUNO,
    COORDENADOR
}
